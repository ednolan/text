namespace boost {
  namespace text {
    template<nf Normalization, typename Char> struct basic_text_view;
    template<nf Normalization, typename Char> 
      text_sort_key 
      collation_sort_key(basic_text< Normalization, Char > const &, 
                         collation_table const &, 
                         collation_flags = collation_flags::none);
    template<nf Normalization1, typename Char1, nf Normalization2, 
             typename Char2> 
      int collate(basic_text< Normalization1, Char1 > const &, 
                  basic_text< Normalization2, Char2 > const &, 
                  collation_table const &, 
                  collation_flags = collation_flags::none);
    template<nf Normalization, typename Char> 
      int operator+(basic_text_view< Normalization, Char > const & t, 
                    Char const * c_str);
  }
